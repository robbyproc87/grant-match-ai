## File: `lib/types/eligibility.ts`

```typescript
import { z } from "zod";

/**
 * GrantEligibility — locked schema for the eligibility JSONB column on `grants`.
 * Stored in Postgres as jsonb. Validated at read-time with Zod.
 *
 * Scoring (0–15, binary):
 *   - 15 if all evaluated hard rules pass
 *   - 0  if any evaluated hard rule fails
 *   - Missing/undefined optional fields = "no constraint" (do not penalize)
 *   - `custom_rules` are NEVER auto-evaluated — surfaced to user as manual checks
 *   - If JSON fails Zod parse: score 0, factor reasoning = "malformed eligibility data"
 */
export const GrantEligibilitySchema = z.object({
  // Hard rule: org must be a registered 501(c)(3). Default: required.
  requires_501c3: z.boolean(),

  // Hard rule: org age in years (inclusive). Optional.
  min_org_years: z.number().int().min(0).optional(),

  // Hard rule: org annual budget bounds in USD. Optional.
  // Inclusive range; either bound can be set independently.
  min_org_budget: z.number().nonnegative().optional(),
  max_org_budget: z.number().nonnegative().optional(),

  // Hard rule: allowed org type categories. If present, org must match at least one.
  allowed_org_types: z
    .array(z.enum(["nonprofit", "school", "government", "tribal"]))
    .optional(),

  // Hard rule: org's geographies must intersect this list (or list contains "national").
  // Use 2-letter US state codes (uppercase) or the literal "national".
  allowed_geographies: z.array(z.string()).optional(),

  // Hard rule: org's geographies must NOT intersect this list.
  // Same format as allowed_geographies.
  excluded_geographies: z.array(z.string()).optional(),

  // Soft rules: human-readable strings shown to the user as a manual checklist.
  // NEVER affect the auto-computed score.
  custom_rules: z.array(z.string()).optional(),
});

export type GrantEligibility = z.infer<typeof GrantEligibilitySchema>;
```
